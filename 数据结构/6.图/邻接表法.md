# 邻接表法(顺序+链式存储)

```c
// 用邻接表存储的图
typedef struct {
  AdjList vertices;
  int vexnum, arcnum;
} ALGraph;

// "边/弧"
typedef struct ArcNodef {
  int adivex;           // 边/弧指向哪个结点
  struct ArcNode *next; // 指向下一条弧的指针
  // InfoType info; // 边权值
} FArcNode;

// 顶点"
typedef struct VNode {
  VertexType data; // 顶点信息
  ArcNode *first;  // 第一条边/弘
} VNode, AdjList[MaxVertexNum];
```
